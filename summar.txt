# Data Summarizer Tool — LLM-powered intelligent data analysis & summary
# Exposed as a Tool for Worker Node via tool_mode=True
#
# What it does:
#   1. Accepts data in ANY format (JSON, markdown tables, text, data_json comment)
#   2. Computes statistics automatically (min, max, mean, totals, distribution)
#   3. Optionally uses the LLM for narrative insights & anomaly detection
#   4. Returns a clean markdown summary with configurable depth

from agentcore.custom import Node
import json
import re
import statistics

SUMMARY_PROMPT = """You are a senior data analyst. Analyze the following dataset and produce a concise executive summary.

**User Request:** {user_request}

**Dataset Overview:**
{stats_block}

**Sample Data (first 10 rows):**
{sample_rows}

**Respond in this exact markdown structure:**

## Summary
<2-3 sentence high-level overview of what this data represents>

## Key Insights
- <insight 1: most important finding with specific numbers>
- <insight 2: trend or pattern>
- <insight 3: notable comparison or outlier>
- <insight 4 if relevant>

## Anomalies / Watch Points
- <anything that looks unusual, missing, or requires attention — or "None detected">

## Recommendation
<1-2 sentences on what action or next step the data suggests>

Keep it concise. Use actual numbers from the data. No waffle."""


QUICK_SUMMARY_PROMPT = """Summarize this dataset in 3-5 bullet points. Be specific with numbers.

**Data:**
{stats_block}

Return ONLY bullet points, no headings."""


# ═══════════════════════════════════════════════════════════════════════════════
# DATA PARSING HELPERS  (identical to DataVisualizer for consistency)
# ═══════════════════════════════════════════════════════════════════════════════

def _try_parse_data_json(text):
    comment_match = re.search(r'<!--\s*data_json:(.*?)\s*-->', text, re.DOTALL)
    if comment_match:
        try:
            data = json.loads(comment_match.group(1).strip())
            if data.get("columns") and data.get("rows"):
                return data, text[comment_match.end():].strip()
        except json.JSONDecodeError:
            pass

    tag_match = re.search(r'<data_json>(.*?)</data_json>', text, re.DOTALL)
    if tag_match:
        try:
            data = json.loads(tag_match.group(1).strip())
            if data.get("columns") and data.get("rows"):
                return data, text[tag_match.end():].strip()
        except json.JSONDecodeError:
            pass

    json_match = re.search(
        r'\{[^{}]*"columns"\s*:\s*\[.*?\].*?"rows"\s*:\s*\[.*?\][^{}]*\}',
        text, re.DOTALL
    )
    if json_match:
        try:
            data = json.loads(json_match.group())
            if data.get("columns") and data.get("rows"):
                return data, text[json_match.end():].strip()
        except json.JSONDecodeError:
            pass

    return None, text


def _try_parse_markdown_table(text):
    lines = text.strip().split("\n")
    table_lines = [l for l in lines if l.strip().startswith("|")]
    if len(table_lines) < 3:
        return None

    header = table_lines[0]
    columns = [c.strip() for c in header.split("|") if c.strip()]
    if not columns:
        return None

    data_start = 1
    for i in range(1, len(table_lines)):
        if re.match(r'^[\s|:-]+$', table_lines[i]):
            data_start = i + 1
        else:
            break

    rows = []
    for line in table_lines[data_start:]:
        cells = [c.strip() for c in line.split("|") if c.strip()]
        if len(cells) != len(columns):
            continue
        parsed_row = []
        for cell in cells:
            clean = cell.replace(",", "").replace("—", "").strip()
            if not clean:
                parsed_row.append(None)
            else:
                try:
                    parsed_row.append(float(clean) if "." in clean else int(clean))
                except ValueError:
                    parsed_row.append(cell)
        rows.append(parsed_row)

    return {"columns": columns, "rows": rows} if rows else None

# STATISTICS ENGINE

def _compute_column_stats(columns, rows):
    """Compute per-column statistics. Returns a list of dicts."""
    col_stats = []
    num_rows = len(rows)

    for i, col_name in enumerate(columns):
        vals = [r[i] for r in rows if i < len(r) and r[i] is not None]
        numeric_vals = []
        for v in vals:
            if isinstance(v, (int, float)):
                numeric_vals.append(float(v))
            else:
                try:
                    numeric_vals.append(float(str(v).replace(",", "")))
                except (ValueError, TypeError):
                    pass

        null_count = num_rows - len(vals)

        if numeric_vals:
            col_stats.append({
                "name": col_name,
                "type": "numeric",
                "count": len(numeric_vals),
                "null_count": null_count,
                "min": min(numeric_vals),
                "max": max(numeric_vals),
                "sum": sum(numeric_vals),
                "mean": statistics.mean(numeric_vals),
                "median": statistics.median(numeric_vals),
                "stdev": statistics.stdev(numeric_vals) if len(numeric_vals) > 1 else 0,
                "unique": len(set(numeric_vals)),
            })
        else:
            str_vals = [str(v) for v in vals if v is not None]
            unique_vals = list(dict.fromkeys(str_vals))  # preserve order, dedupe
            col_stats.append({
                "name": col_name,
                "type": "text",
                "count": len(str_vals),
                "null_count": null_count,
                "unique": len(unique_vals),
                "top_values": unique_vals[:5],
            })

    return col_stats


def _format_number(v):
    """Format a number for display."""
    if abs(v) >= 1_000_000_000:
        return f"{v / 1_000_000_000:.2f}B"
    if abs(v) >= 1_000_000:
        return f"{v / 1_000_000:.2f}M"
    if abs(v) >= 1_000:
        return f"{v:,.0f}"
    if abs(v) < 0.01 and v != 0:
        return f"{v:.4f}"
    return f"{v:,.2f}"


def _build_stats_block(columns, rows, col_stats):
    """Build a human-readable stats block for the prompt."""
    lines = [
        f"Total rows: {len(rows)}",
        f"Total columns: {len(columns)}",
        "",
    ]
    for cs in col_stats:
        if cs["type"] == "numeric":
            lines.append(
                f"  {cs['name']} [numeric]: "
                f"min={_format_number(cs['min'])}, "
                f"max={_format_number(cs['max'])}, "
                f"mean={_format_number(cs['mean'])}, "
                f"median={_format_number(cs['median'])}, "
                f"sum={_format_number(cs['sum'])}, "
                f"stdev={_format_number(cs['stdev'])}, "
                f"nulls={cs['null_count']}"
            )
        else:
            top = ", ".join(f'"{v}"' for v in cs["top_values"])
            lines.append(
                f"  {cs['name']} [text]: "
                f"{cs['unique']} unique values, "
                f"top: [{top}], "
                f"nulls={cs['null_count']}"
            )
    return "\n".join(lines)


def _build_heuristic_summary(columns, rows, col_stats, detail_level):
    """Build a pure-stats summary without an LLM. Always fast."""
    num_rows = len(rows)
    num_cols = len(columns)
    parts = []

    # ── Overview ──
    parts.append(f"## Dataset Overview")
    parts.append(f"- **{num_rows:,} rows** × **{num_cols} columns**")

    numeric_cols = [cs for cs in col_stats if cs["type"] == "numeric"]
    text_cols    = [cs for cs in col_stats if cs["type"] == "text"]
    parts.append(f"- **{len(numeric_cols)} numeric** column(s), **{len(text_cols)} text** column(s)")

    total_nulls = sum(cs["null_count"] for cs in col_stats)
    if total_nulls:
        parts.append(f"- ⚠️ **{total_nulls} missing value(s)** across all columns")
    else:
        parts.append(f"- ✅ No missing values")

    # ── Numeric column stats ──
    if numeric_cols:
        parts.append(f"\n## Numeric Columns")
        header = "| Column | Min | Max | Mean | Median | Sum | Std Dev |"
        sep    = "|--------|-----|-----|------|--------|-----|---------|"
        rows_md = []
        for cs in numeric_cols:
            rows_md.append(
                f"| {cs['name']} "
                f"| {_format_number(cs['min'])} "
                f"| {_format_number(cs['max'])} "
                f"| {_format_number(cs['mean'])} "
                f"| {_format_number(cs['median'])} "
                f"| {_format_number(cs['sum'])} "
                f"| {_format_number(cs['stdev'])} |"
            )
        parts.append("\n".join([header, sep] + rows_md))

        # Highlight highest-value numeric column
        if detail_level != "brief":
            max_col = max(numeric_cols, key=lambda c: c["sum"])
            parts.append(
                f"\n> **{max_col['name']}** has the largest total: "
                f"**{_format_number(max_col['sum'])}** "
                f"(mean {_format_number(max_col['mean'])}, "
                f"range {_format_number(max_col['min'])}–{_format_number(max_col['max'])})"
            )

    # ── Text column stats ──
    if text_cols and detail_level != "brief":
        parts.append(f"\n## Categorical Columns")
        for cs in text_cols:
            top = ", ".join(f"`{v}`" for v in cs["top_values"][:3])
            parts.append(
                f"- **{cs['name']}**: {cs['unique']:,} unique values — top: {top}"
            )

    # ── Distribution for primary numeric column (top 5 by value) ──
    if numeric_cols and detail_level == "detailed" and text_cols:
        parts.append(f"\n## Top 5 Rows by {numeric_cols[0]['name']}")
        num_col_idx = columns.index(numeric_cols[0]["name"])
        sorted_rows = sorted(
            rows,
            key=lambda r: float(str(r[num_col_idx]).replace(",", "")) if num_col_idx < len(r) and r[num_col_idx] is not None else 0,
            reverse=True
        )[:5]
        text_col_idx = columns.index(text_cols[0]["name"])
        for r in sorted_rows:
            label = str(r[text_col_idx])[:30] if text_col_idx < len(r) else "?"
            value = _format_number(float(str(r[num_col_idx]).replace(",", ""))) if num_col_idx < len(r) and r[num_col_idx] is not None else "—"
            parts.append(f"- **{label}**: {value}")

    return "\n".join(parts)

class DataSummarizerNode(Node):
    display_name = "Data Summarizer"
    description = (
        "Use this tool to summarize, analyze, or describe a dataset. "
        "Accepts data in JSON, markdown table, or text format. "
        "Returns key statistics, insights, anomalies, and recommendations. "
        "Connect an LLM for narrative analysis; works without one for pure stats."
    )
    icon = "table-2"
    name = "DataSummarizer"

    inputs = [
        MessageTextInput(
            name="input_value",
            display_name="Data + Instructions",
            info="Data (JSON, markdown table, or text) + optional summarization instructions.",
            tool_mode=True,
        ),
        HandleInput(
            name="llm",
            display_name="Language Model",
            input_types=["LanguageModel"],
            info="Optional LLM for narrative insights and anomaly detection. "
                 "If not connected, returns pure statistics only.",
            required=False,
        ),
        DropdownInput(
            name="detail_level",
            display_name="Detail Level",
            options=["brief", "standard", "detailed"],
            value="standard",
            info=(
                "'brief' = 3-5 bullet points. "
                "'standard' = stats table + key insights. "
                "'detailed' = full executive report with top-N breakdown."
            ),
        ),
        DropdownInput(
            name="summary_mode",
            display_name="Summary Mode",
            options=["auto", "stats_only", "llm_only"],
            value="auto",
            info=(
                "'auto' = stats first, LLM narrative if connected. "
                "'stats_only' = always pure statistics, never calls LLM. "
                "'llm_only' = only LLM narrative (requires LLM)."
            ),
        ),
    ]

    outputs = [
        Output(display_name="Summary", name="output", method="build_output"),
    ]

    def build_output(self) -> Message:
        raw_input = self.input_value or ""
        if not raw_input.strip():
            return Message(text="No data provided. Pass data to summarize.")

        # ── Step 1: Parse data ──
        data = None
        instructions = raw_input

        parsed, remaining = _try_parse_data_json(raw_input)
        if parsed:
            data = parsed
            instructions = remaining or "summarize this data"

        if not data:
            parsed = _try_parse_markdown_table(raw_input)
            if parsed:
                data = parsed
                non_table = [l for l in raw_input.split("\n") if not l.strip().startswith("|")]
                instructions = "\n".join(non_table).strip() or "summarize this data"

        if not data:
            try:
                raw_json = json.loads(raw_input.strip())
                if isinstance(raw_json, dict) and raw_json.get("columns") and raw_json.get("rows"):
                    data = raw_json
                    instructions = "summarize this data"
            except (json.JSONDecodeError, ValueError):
                pass

        if not data or not data.get("columns") or not data.get("rows"):
            return Message(
                text="Could not extract data from the input. "
                     "Please provide a data table, JSON with `columns`/`rows`, "
                     "or data from the Data Visualizer output."
            )

        columns = data["columns"]
        rows    = data["rows"]

        # ── Step 2: Compute statistics ──
        col_stats = _compute_column_stats(columns, rows)
        stats_block = _build_stats_block(columns, rows, col_stats)

        # ── Step 3: Build output based on mode ──
        output_parts = []

        use_llm   = self.summary_mode != "stats_only" and self.llm and self.llm != ""
        use_stats = self.summary_mode != "llm_only"

        if self.summary_mode == "llm_only" and (not self.llm or self.llm == ""):
            return Message(text="⚠️ `llm_only` mode requires a Language Model to be connected.")

        # Stats block (always shown unless llm_only)
        if use_stats:
            stats_md = _build_heuristic_summary(columns, rows, col_stats, self.detail_level)
            output_parts.append(stats_md)

        # LLM narrative
        if use_llm:
            try:
                sample_rows = rows[:10]
                sample_text = "\n".join(
                    "  " + " | ".join(str(v) if v is not None else "—" for v in r)
                    for r in sample_rows
                )

                if self.detail_level == "brief":
                    prompt = QUICK_SUMMARY_PROMPT.format(stats_block=stats_block)
                else:
                    prompt = SUMMARY_PROMPT.format(
                        user_request=instructions,
                        stats_block=stats_block,
                        sample_rows=f"Columns: {columns}\n{sample_text}",
                    )

                response = self.llm.invoke(prompt)
                narrative = response.content if hasattr(response, "content") else str(response)
                narrative = narrative.strip()

                if use_stats:
                    output_parts.append("\n---\n## AI Analysis")
                output_parts.append(narrative)
            except Exception as e:
                output_parts.append(f"\n*LLM analysis unavailable: {e}*")

        # Embed data_json for follow-up chart requests
        data_json_obj = {"columns": columns, "rows": rows}
        data_comment = f"\n<!-- data_json:{json.dumps(data_json_obj)} -->"
        output_parts.append(data_comment)

        self.status = f"{len(rows):,} rows × {len(columns)} cols | {self.detail_level}"
        return Message(text="\n\n".join(output_parts))
